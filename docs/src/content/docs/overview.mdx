import { Card, CardGrid } from '@astrojs/starlight/components';

Lattice turns a plain Markdown file into a boardroom-ready slide deck:
you write the words, pick a named component per slide, and the engine
sets every page in one shared palette. For the full story — what it is,
who it's for, and what it deliberately isn't — start at
[What is Lattice?](/introduction/) This page is the map.

<CardGrid>
	<Card title="Text in, PDF out" icon="document">
		No dragging, no nudging, no alignment guides. Write the words, pick a
		component, build.
	</Card>
	<Card title="One palette, every deck" icon="setting">
		Brand colors live in one file. Change them once; every deck picks
		them up on the next build.
	</Card>
	<Card title="Themed diagrams" icon="puzzle">
		Mermaid flowcharts and more render in the deck's palette
		automatically — no per-diagram styling.
	</Card>
	<Card title="Version control" icon="seti:git">
		A deck is text. A normal `git diff` shows what changed between
		revisions, the way it does for code.
	</Card>
	<Card title="Accessible by default" icon="approve-check">
		Contrast is WCAG AA across every component.
	</Card>
	<Card title="No service, no account" icon="laptop">
		Runs on your laptop or in CI. Fully offline-capable. No telemetry.
		MIT-licensed.
	</Card>
</CardGrid>
